Logical Operators



Logical Operation
AND : SELECT * FROM EMPLOYEES WHERE SALARY<1000 AND AGE>25;
OR : SELECT * FROM STUDENTS WHERE RANK>20 OR MARK>600;
NOT : SELECT * FROM EMP WHERE NOT SALARY=10000;



GROUP -ing Operations
SELECT FName, LName FROM EMP GROUP BY Name;



Aggregate Functions ( Mathematical Operations )
AVERAGE : SELECT avg(salary) FROM EMP;
MINIMUM : SELECT min(salary) FROM EMP;
MAXIMUM : SELECT max(salary) FROM EMP;
SUM : SELECT sum(salary) FROM EMP;
CONUT : SELECT count(id) FROM EMP;
DISTINCT : SELECT distinct(name) FROM EMP;
ROUND : SELECT ROUND (Marks,2) FROM STUDENTS;



String Functions ( String Handling Operations )
LENGTH : SELECT LEN(‘APPLE’) ;
REVERSE : SELECT REVERSE(‘APPLE’) ;
REPLICATE : SELECT REPLICATE (‘APPLE’ ,S) ;
UPPERCASE : SELECT UPPER (‘apple’) ;
LOWERCASE : SELECT LOWER (‘APPLE’) ;
CONCATENATION : SELECT CONCAT (‘NE’,’PA’,’L’) ;
LEFTMOST CHARACTER : SELECT LEFT (‘APPLE’,3) ;
RIGHTMOST CHARACTER : SELECT RIGHT (‘APPLE’,3) ;



Others
IN : SELECT ADDRESS FROM STD WHERE IN (“KTM”, “BKT”,”PTN”);
BETWEEN : SELECT SALARY FROM EMP BETWEEN 30000 AND 40000;
PERCENTAGE : SELECT NAME FROM STD LIKE “%a”;
UNDERSCORE : SELECT NAME FROM STD LIKE “%_b”;
HAVING : SELECT NAME FROM STD WHERE GRADE=”A” HAVING MARKS>800;
